// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

import "@openzeppelin/contracts/token/ERC721/ERC721.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

contract CarNFT is ERC721, Ownable {
    struct CarInfo {
        string brand;
        string model;
        uint256 year;
        uint256 mileage;
        string[] serviceRecords;
    }

    mapping(uint256 => CarInfo) public carInfos;
    uint256 private _nextTokenId;

    event CarMinted(uint256 indexed tokenId, address indexed owner);
    event ServiceRecordAdded(uint256 indexed tokenId, string record);

    constructor() ERC721("CarLife", "CAR") {}

    function mintCar(
        string memory brand,
        string memory model,
        uint256 year
    ) external returns (uint256) {
        uint256 tokenId = _nextTokenId++;
        _safeMint(msg.sender, tokenId);

        carInfos[tokenId] = CarInfo({
            brand: brand,
            model: model,
            year: year,
            mileage: 0,
            serviceRecords: new string[](0)
        });

        emit CarMinted(tokenId, msg.sender);
        return tokenId;
    }

    function addServiceRecord(uint256 tokenId, string memory record) external {
        require(_ownerOf(tokenId) == msg.sender, "Not owner");
        carInfos[tokenId].serviceRecords.push(record);
        emit ServiceRecordAdded(tokenId, record);
    }

    function updateMileage(uint256 tokenId, uint256 newMileage) external {
        require(_ownerOf(tokenId) == msg.sender, "Not owner");
        require(newMileage >= carInfos[tokenId].mileage, "Invalid mileage");
        carInfos[tokenId].mileage = newMileage;
    }
}
